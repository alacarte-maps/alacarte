137 uint64_t xy2hilbert(FixedPoint p)
138 {
139  // convert signed to unsigned with offset

141  uint32_t y = p.y - std::numeric_limits<coord_t>::min();

152  for (int k = 1; k < r; k++) {

154  v0 = ((v0 & (v1 ^ notx)) | (~v0 & (v1 ^ noty))) >> 1;
